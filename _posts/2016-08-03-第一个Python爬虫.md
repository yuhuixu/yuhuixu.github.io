爬取[快代理](http://www.kuaidaili.com/proxylist/)网站免费的IP代理地址

## py代码

```python
#coding=utf-8

import urllib2
import re #导入正则表达式模块

'''
http://www.kuaidaili.com/proxylist/1/
'''

def geturl():
    p = re.compile(r'<td data-title="IP">(.+?)</td>')

    for i in range(1,10):
        target = r"http://www.kuaidaili.com/proxylist/%d/"%i
        print target
        #打开目标网页
        req = urllib2.urlopen(target)
        #读取内容
        result = req.read()
        #匹配正则表达式
        matchs = p.findall(result)
        print matchs

geturl()
```
